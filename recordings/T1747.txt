import re
from playwright.sync_api import Playwright, sync_playwright, expect


def run(playwright: Playwright) -> None:
    browser = playwright.chromium.launch(headless=False)
    context = browser.new_context()
    page = context.new_page()
    page.goto("https://release.pear.us/yu-xiao")
    page.get_by_role("button", name="0", exact=True).click()
    page.get_by_role("link", name="Create a Post").click()
    page.get_by_role("button", name="Close").click()
    page.get_by_role("tab", name="Content").click()
    page.get_by_role("textbox", name="Pick a bold title for your").click()
    page.get_by_role("textbox", name="Pick a bold title for your").fill("test")
    page.get_by_role("textbox", name="Enter a subtitle to enrich").click()
    page.get_by_role("textbox", name="Enter a subtitle to enrich").fill("123")
    page.get_by_role("button", name="Save as draft").click()
    page.get_by_text("Post saved as draft!").click()

    # ---------------------
    context.close()
    browser.close()


with sync_playwright() as playwright:
    run(playwright)
