import re
from playwright.sync_api import Playwright, sync_playwright, expect


def run(playwright: Playwright) -> None:
    browser = playwright.chromium.launch(headless=False)
    context = browser.new_context()
    page = context.new_page()
    page.goto("https://release.pear.us/yu-xiao")
    page.locator("div").filter(has_text=re.compile(r"^Auto test form\.$")).first.click()
    page.locator("input[name=\"Note\"]").click()
    page.locator("input[name=\"Note\"]").fill("test auto send")
    page.locator("textarea[name=\"Meseege\"]").click()
    page.locator("textarea[name=\"Meseege\"]").fill("test auto")
    page.locator("textarea[name=\"Meseege\"]").press("ControlOrMeta+a")
    page.get_by_text("test auto").press("ControlOrMeta+c")
    page.get_by_text("test auto").fill("test auto ")
    page.get_by_text("test auto").press("ControlOrMeta+a")
    page.get_by_text("test auto").press("ControlOrMeta+c")
    page.get_by_text("test auto").fill("test auto test auto test auto test auto test auto test auto test auto test auto test auto test auto test auto test auto ")
    page.get_by_role("textbox", name="Enter your email address").click()
    page.get_by_role("textbox", name="Enter your email address").fill("yuxiao.zhu.ext@1m.app")
    page.get_by_role("textbox", name="Phone number").click()
    page.get_by_role("textbox", name="Phone number").fill("+13018427082")
    page.get_by_role("button", name="Submit").click()
    page.get_by_text("The form was submitted").click()

    # ---------------------
    context.close()
    browser.close()


with sync_playwright() as playwright:
    run(playwright)
